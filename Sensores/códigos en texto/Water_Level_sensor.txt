#include "mbed.h"              
 
//Serial pc(USBTX, USBRX); // ignorar esto
DigitalIn water(PTA1);

int main() {
        SIM->SCGC5 |= 0x400; 
        PORTB->PCR[18] = 0x100; 
        PORTB->PCR[19] = 0x100; 
        PTB->PDDR |= 0xC0000; 
        PTB->PDOR |= 0xC0000; 
        SIM->SCGC5 |= 0x200; 
while(1){
    int sensor = water.read();
    //pc.printf("%d",sensor); /*ignorar esto
    if(sensor == 1){
        PTB->PDOR &= ~0x80000; 
        wait(5);
        }
    else if(sensor == 0){
            PTB->PDOR |= 0x80000; 
        }
    }
}
